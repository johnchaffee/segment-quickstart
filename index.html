Segment Quickstart

Identify

  analytics.identify('f4ca124298', {
    name: 'Michael Brown',
    email: 'mbrown@example.com'
  });

  analytics.identify(' {{user.id}} ', {
    name: '{{user.fullname}}',
    email: '{{user.email}}'
  });

Track Actions

  analytics.track('Signed Up', {
    plan: 'Enterprise'
  });
  
  analytics.track('Article Bookmarked', {
    title: 'Snow Fall',
    subtitle: 'The Avalanche at Tunnel Creek',
    author: 'John Branch'
  });
